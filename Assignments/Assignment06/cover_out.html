}

func INFO(shape SHAPE) {
        fmt.Println(shape.AREA())
}
func (square SQUARE) AREA() float32 {
        return float32(square.length * square.length)
}
func (circle CIRCLE) AREA() float32 {
        return PI * float32(circle.radius*circle.radius)
}
